Apuntes sobre: Git y Github

Basicos

                            
                            #Inicializa el repositorio
                            git init
                            #Verificar si hay cambios
                            git status
                            #Agrega archivos
                            git add ruta_de_archivo
                            #Agregar directorio
                            git add ruta_de_directorio
                            #Agregar todos
                            git add .
                            #Ver diferencias entre staged y local
                            git diff
                            #Remover todos los cambios de staged
                            git reset HEAD
                            #Remover el archivo del area de staging
                            git reset HEAD ruta_de_archivo
                            #Remover el tracking por parte de git
                            git rm --cached ruta_de_archivo
                            
                        
                            
                                #Agregar cambios al repositorio
                                git commit -m "Mensaje de commit"
                                #Agregar cambios a staging y guardar cambios
                                git commit -a -m "Mensaje de commit"
                                #Reparar el ultimo commit a√±adiendole cambios
                                git commit --amend -m "Mensaje de commit"
                                #Verificar historial de commits
                                git log
                                #Verificar cambios especificos del ultimo commit
                                git log --stat
                                #Verificar historial de commits de forma grafica
                                git log --graph
                                #historial grafico y detallado
                                git log --all --graph --decorate --oneline
                                #historial completo de acciones del repositorio
                                git reflog
                                #Abrir consola grafica del repositorio
                                gitk --all

Branchs / Ramas 

                        
                            #Verificar en que rama estas trabajando
                            git branch
                            #Crear una nueva rama
                            git branch nombre_de_rama
                            #Renombrar una rama
                            git branch -m nombre_de_rama nuevo_nombre_de_rama
                            #Cambiar de rama
                            git checkout nombre_de_rama
                            #Crear una nueva rama y cambiar a ella
                            git checkout -b nombre_de_rama
                            #Restarurar archivos de un commit anterior
                            git checkout nombre_de_commit -- archivo
                            #Eliminar una rama
                            git branch -d nombre_de_rama
                            #Fusionar Ramas (Debe hacerse desde la rama base)
                            git merge nombre_de_rama

Practicas Delicadas

(Pueden afectar la integridad del repositorio)

                        
                        #Deshacer el ultimo commit
                        git reset --hard HEAD~1
                        #Deshacer un commit especifico
                        git reset --hard nombre_de_commit
                        #Reescribir el historial de una rama
                        git rebase -i nombre_de_rama
                        #Tomar los cambios de un commit especificio e integrarlo a una rama
                        git cherry-pick nombre_de_commit
                        #Si por alguna razon quieres abortar la integracion de dicho commit
                        git cherry-pick --abort
                        #Ver estado anterior de archivos
                        git checkout commit_hash archivo
                        #Si se ejecuta un commit luego de este comando se perderan todos los cambios
                        #posteriores de la rama

Comandos Utiles

                        
                        #Versionar por tags
                        git tag -a nombre-tag -m "mensaje" commit_hash
                        #Ej: git tag -a V0.1 -m "Version para despliegue 0.1" 60b91c2
                        #Listar tags
                        git tag -l
                        #Listar tag y sus referencias
                        git show-ref --tags
                        #Borrar un tag
                        git tag -d nombre-tag
                        #Borrar un tag en repositorio remoto
                        git push repo_remoto :refs/tags/nombre-tag
                        #Borrar archivos no deseados
                        git clean --dry-run #lista todos los archivos no trackeados a borrar
                        git clean -f #borra todos los archivos no trackeados

                        #Los archivos agregados en el .gitignore no seran afectados por este comando
                        #si se copia un directorio con archivos trackeados tampoco se vera afectado por este comando
                        
                        #Busqueda en archivos
                        
                        git grep "palabra" #Busca la palabra en todos los archivos
                        git grep -n "palabra" #Retorna la linea del archivo esta lo que estamos buscando
                        git grep -c "palabra" #Retorna cuantas veces se repite lo que buscamos y en que archivo
                        
                        //Busqueda de commits
                        git log -S palabra (busca en los commits si usamos la palabra)

Repositorios Remotos & Stash

                            
                            #Clonar repositorio
                            git clone url-repositorio
                            #A√±adir repositorio remoto HTTPS
                            git remote add nombre-repositorio url-repositorio
                            #Traer cambios de remoto a local
                            git pull nombre-repositorio nombre-rama
                            #De remoto a local sin histrias relacionadas
                            git pull nombre-repositorio nombre-rama --allow-unrelated-histories
                            #Hacer push de una rama a repo remoto
                            git push repo_remoto nombre-rama

                            #Uso de Stash

                            #Guardar cambios de forma temporal#cambios flotantes
                            git stash
                            git stash save "Mensaje" #guardamos el stash con un mensaje
                            git stash list #lista todos los stash que tengas
                            git stash pop #trae el ultimo stash y lo integra a la rama
                            git stash apply stash@{n} #trae el stash que necesites aplicando el numero dentro de las llaves y lo aplica a la rama
                            git stash drop #borra el ultimo stash
                            git stash clear #borra todos los stash
                            git stash branch nombre_de_la_rama #Para colocar los cambios en una nueva rama
